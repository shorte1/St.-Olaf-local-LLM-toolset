Here's a Python Flask implementation of a REST API endpoint that accepts a JSON body containing a username and password, validates both fields, and returns a JWT token:

```python
from flask import Flask, request, jsonify, render_template
import uuid
import bcrypt
from flask_jose import JWTError, jwt
from flask_cors import CORS

# Initialize Flask app
app = Flask(__name__)

# Setup secret keys (update these in production)
SECRET_KEY = 'your-secret-key-here'
TOKEN_SECRET_KEY = 'another-secret-key-here'

# Configure JWT settings
app.config['JWT_SECRET_KEY'] = TOKEN_SECRET_KEY
app.config['JWT issuer'] = 'YourApp'
app.config['JWT audience'] = 'http://localhost:5000'

# Initialize other components
jwt.init_app(app)
CORS(app)

@app.route('/', methods=['POST'])
def auth():
    # Get the JSON request body
    data = request.json

    # Validate required fields
    if not all(key in data for key in ['username', 'password']):
        return jsonify({'error': 'Username and password are required'}), 400

    username = data['username'].strip()
    password = data['password'].strip()

    if not username or not password:
        return jsonify({'error': 'Username and password must be provided'}), 400

    # Hash the password
    hashed_password = bcrypt.hashpw(password.encode('utf-8'), bcrypt.gensalt())

    try:
        # Generate JWT token
        token = jwt.generate_token(
            {'username': username},
            issuer=app.config['JWT issuer'],
            audience=app.config['JWT audience']
        )
        
        return jsonify({'access_token': token.decode('ascii')}),
                200
        
    except Exception as e:
        return jsonify({'error': str(e)}), 500
```
