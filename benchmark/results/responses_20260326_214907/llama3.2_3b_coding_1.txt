def sieve_of_eratosthenes(n):
    """
    This function uses the Sieve of Eratosthenes algorithm to find all prime numbers up to n.

    Args:
        n (int): The upper limit for finding prime numbers.

    Returns:
        list: A list of all prime numbers up to n.
    """

    # Create a boolean array, prime, of size n+1
    prime = [True] * (n + 1)
    prime[0] = prime[1] = False

    # Iterate from 2 to sqrt(n)
    for p in range(2, int(n ** 0.5) + 1):
        # If p is a prime number
        if prime[p]:
            # Mark as composite all the multiples of p
            for i in range(p * p, n + 1, p):
                prime[i] = False

    # Return a list of all prime numbers up to n
    return [p for p in range(2, n + 1) if prime[p]]
