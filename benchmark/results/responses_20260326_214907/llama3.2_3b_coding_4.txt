2. It iterates over each element `i` in the input list using the outer loop.

5. Finally, it returns the `result` list.

def process(data):
    """
    Returns pairs of indices where the sum of corresponding elements in the input list is equal to zero.

    Args:
        data (list): A list of numbers.

    Returns:
        list: Pairs of indices where the sum is zero.
    """
    if not isinstance(data, list) or len(data) == 0:
        raise ValueError("Input must be a non-empty list")
    for i in range(len(data)):
        for j in range(i + 1, len(data)):  # Start inner loop from i+1
            if data[i] + data[j] == 0:
                yield (i, j)  # Use yield instead of append to reduce memory usage

# Example usage
data = [1, -2, 3, -4]
result = list(process(data))
print(result)
