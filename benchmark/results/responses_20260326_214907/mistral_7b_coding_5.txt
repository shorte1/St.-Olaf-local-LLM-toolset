Prompt: Write a REST API endpoint in Python using Flask that accepts a JSON body with a username and password, validates that neither is empty, and returns a JWT token.
========================================
 To create a REST API endpoint in Python using Flask for authentication, you'll need to install Flask and PyJWT (for generating the JWT token). You can do this by running:

```bash
pip install flask pyjwt
```

After installing the necessary libraries, you can create a simple Flask application with an authentication endpoint as follows:

```python
from flask import Flask, request, jsonify
from jwt import encode, decode, ExpiredSignatureError
import datetime
import hashlib
import secrets

app = Flask(__name__)

SECRET_KEY = 'your-secret-key'  # Replace with a secure secret key
ALGORITHM = 'HS256'

def generate_password_hash(password):
    salt = hashlib.sha256(secrets.token_bytes(16)).hexdigest()
    return hashlib.pbkdf2_hmac("sha256", password.encode(), salt.encode(), 100000)

def verify_password_hash(hashed_password, password):
    salt = hashed_password[:64]
    stored_password = hashed_password[64:]
    return hashlib.pbkdf2_hmac("sha256", password.encode(), salt.encode(), 100000) == stored_password

def authenticate(username, password):
    user = {
        "username": username,
        "password": generate_password_hash('initial-password')  # Replace with your hashed initial password for users
    }

    return user['username'] == username and verify_password_hash(user['password'], password)

def create_jwt_token(data):
    to_encode = data.copy()
    expiration = datetime.datetime.utcnow() + datetime.timedelta(minutes=30)
    to_encode["exp"] = expiration
    encoded_jwt = encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)
    return encoded_jwt

@app.route('/token', methods=['POST'])
def token():
    if not request.is_json or 'username' not in request.json or 'password' not in request.json:
        return jsonify({"error": "Invalid request"}), 400

    username = request.json['username']
    password = request.json['password']

    if not username or not password:
        return jsonify({"error": "Both username and password are required"}), 400

    if authenticate(username, password):
        jwt_token = create_jwt_token({"user_id": username})
        return jsonify({"token": jwt_token}), 200
    else:
        return jsonify({"error": "Invalid credentials"}), 401

if __name__ == "__main__":
    app.run(debug=True)
```

In this example, the `/token` endpoint accepts a POST request with JSON data containing a username and password. It validates that neither is empty, authenticates the user by comparing the provided password to the hashed stored password (using a simple salted hash), generates a JWT token if successful, and returns it as part of the response.
