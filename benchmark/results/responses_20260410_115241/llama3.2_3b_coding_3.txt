Prompt: Write a Python function that takes a list of integers and returns the two numbers that add up to a target sum. It should run in O(n) time.
========================================
def two_sum(nums, target):
    num_dict = {}
    
    for i, num in enumerate(nums):
        complement = target - num
        if complement in num_dict:
            return [num_dict[complement], i]
        else:
            num_dict[num] = i
    
    return None

# Example usage
nums = [2, 7, 11, 15]
target_sum = 9
result = two_sum(nums, target_sum)
if result:
    print(f"Numbers that add up to {target_sum}: {result[0]} and {result[1]}")
else:
    print("No solution found")
