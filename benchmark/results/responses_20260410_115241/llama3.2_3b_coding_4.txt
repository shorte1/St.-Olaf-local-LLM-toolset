def process(data):
    """
    Find all pairs of indices in the input list where the sum of the elements at those indices equals zero.

    Args:
        data (list): A list of integers.

    Returns:
        list: A list of tuples, where each tuple contains a pair of indices that satisfy the condition.
    """

        raise ValueError("Input must be a non-empty list")

    for num in data:

            raise ValueError("List must contain only integers")

    result = set()  # Use a set to automatically eliminate duplicates
    for i, num1 in enumerate(data):
        for j, num2 in enumerate(data):
